filterQueryParams

  • filterQueryParams(key: string, pred: (s: string) => boolean, options?: { limit?: number }): QueryMutation
    • Defined in filterQueryParams.ts:18

    Filters query parameters for the given key using the given pred. If limit option is specified, the excess values will be ommited.

    Other parameters remain as they were.

    example
    const query = {
      a: ["abb", "axa", "aaa"],
      b: "bbb"
    }
    filterQueryParams("a", (s) => s.endsWith("a"), { limit: 1 })(query)
    // -> { a: ["axa"], b: "bbb" }

getMultipleQueryParams

  • getMultipleQueryParams(query: ParsedUrlQuery, key: string, pred?: (s: string) => boolean): string[]
    • Defined in getMultipleQueryParams.ts:22

    Returns an array of the values for the specified key in the query object. If pred specified, returns the values that meet it.

    example
    getMultipleQueryParams({},            "id") === []
    getMultipleQueryParams({ id: "aaa" }, "id") === ["aaa"]

    // with pred specified
    const is_a = (s: string) => s === "a"
    getMultipleQueryParams({ id: "a" },        "id", is_a) === ["a"]
    getMultipleQueryParams({},                 "id", is_a) === []
    getMultipleQueryParams({ id: "b" },        "id", is_a) === []
    getMultipleQueryParams({ id: ["b", "a"] }, "id", is_a) === ["a"]

removeQueryParams

  • removeQueryParams(options: Record<string, undefined | null | string | boolean | string[]>): QueryMutation
    • Defined in removeQueryParams.ts:41

    By passing ({ key0: predicate0, key1: predicate1, ... }) style object, removes values for key from query object.

    You can pass predicates from one of them;

    1. true — to remove all for the key

    const query = { a: ["aaa", "abb"] }
    removeQueryParams({ a: true })(query)
    // -> { a: [] }

    2. false, undefined, null, or "" — not to remove anything for the key

    const query = { a: ["aaa", "abb"] }
    removeQueryParams({ a: false })(query)
    // -> { a: ["aaa", "abb"] }

    3. single string — to remove it

    const query = { a: ["aaa", "abb"] }
    removeQueryParams({ a: "aaa" })(query)
    // -> { a: ["abb"] }

    4. array of string — to remove all of them

    const query = { a: ["aaa", "abb", "acc"] }
    removeQueryParams({ a: ["aaa", "abb"] })(query)
    // -> { a: ["acc"] }

    Parameters

    • options: Record<string, undefined | null | string | boolean | string[]>

    Returns QueryMutation

resetQuery

  • resetQuery(options?: { ignore?: string | (undefined | null | string | number | boolean)[] }): QueryMutation
    • Defined in resetQuery.ts:18

    Resets the query (into empty object).

    Keys specified in ignore option will not be deleted.

    const query = { a: "aaa", b: "bbb" }
    resetQuery({ ignore: "a" })(query) // -> { a: "aaa" }
    resetQuery({ ignore: ["a", "meaningless"] }) // -> { a: "aaa" }

    Falsy value or boolean values in the array will be discarded.

    resetQuery({ ignore: ["a", true, false, 0, null] })(query) // -> { a: "aaa" }

    Parameters

    • options: { ignore?: string | (undefined | null | string | number | boolean)[] } = {}
      • Optional ignore?: string | (undefined | null | string | number | boolean)[]

        keys not to delete. Falsy value or true will be discarded.

setQueryParams

  • setQueryParams(params: Record<string, undefined | null | string | false | 0 | string[]>): QueryMutation
    • Defined in setQueryParams.ts:26

    Sets values of query parameters.

    for each key in params object,

    • if value is falsy ... resets the values for the key.
    • otherwise ... replace old values for the key with the given value
    example
    const query = { a: "aaa" }

    setQueryParams({ a: "abc" })(query) // -> { a: "abc" }
    setQueryParams({ b: "bbb" })(query) // -> { a: "aaa", b: "bbb" }
    setQueryParams({})(query) // -> { a: "aaa" }

    // falsy values like: 0 | null | undefined | false | ""
    setQueryParams({ a: null })(query) // -> { a: [] }

    Parameters

    • params: Record<string, undefined | null | string | false | 0 | string[]>

    Returns QueryMutation
